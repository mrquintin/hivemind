Step 2 — Add search function to scraper service:

  Add:
  MAX_SEARCH_RESULTS = 5

  def search_and_scrape(query: str, max_results: int = MAX_SEARCH_RESULTS) -> ScrapeResult:
      from duckduckgo_search import DDGS
      with DDGS() as ddgs:
          results = list(ddgs.text(query, max_results=max_results))
      texts = []
      total_bytes = 0
      for r in results:
          url = r.get("href", "")
          if not url:
              continue
          try:
              result = scrape_url_cached(url)
              texts.append(f"--- Source: {url} ---\n{result.text}")
              total_bytes += result.byte_size
          except Exception:
              texts.append(f"--- Source: {url} ---\n[Failed to fetch]")
      combined = "\n\n".join(texts)
      truncated = len(combined) > MAX_TEXT_CHARS
      if truncated:
          combined = combined[:MAX_TEXT_CHARS]
      return ScrapeResult(
          text=combined,
          title=f"Search: {query}",
          byte_size=total_bytes,
          truncated=truncated,
      )
